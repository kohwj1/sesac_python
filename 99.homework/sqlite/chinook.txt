SELECT CONCAT(e.FirstName, ' ', e.LastName) AS AgentName, i.InvoiceId FROM invoices i JOIN customers c JOIN employees e WHERE e.EmployeeId = c.SupportRepid AND c.CustomerId = i.CustomerId;
SELECT i.Total, CONCAT(c.FirstName, ' ', c.LastName) AS CustomerName, c.Country, CONCAT(e.FirstName, ' ', e.LastName) AS EmployeeName FROM invoices i JOIN customers c JOIN employees e WHERE i.CustomerId = c.CustomerId AND c.SupportRepid = e.EmployeeId;
